33 template <class T, int Rows, int Columns, eMatrixOrder Order, eMatrixLayout Layout, bool Packed>
34 auto DecomposeQR(Matrix<T, Rows, Columns, Order, Layout, Packed> m) {
35  static_assert(Rows >= Columns);
36 
37  Matrix<T, Rows, Rows, Order, Layout, Packed> Q;
38  Matrix<T, Rows, Columns, Order, Layout, Packed> R;
39 
40  R = m;
41  Q = Identity();
42 
43  Matrix<T, Rows, Rows, Order, Layout, Packed> Qi;
44  Vector<T, Rows, Packed> u;
45  Matrix<T, Rows, 1, Order, eMatrixLayout::ROW_MAJOR, Packed> v;
46 
47  for (int col = 0; col < m.ColumnCount(); ++col) {
48  u = R.Column(col);
49  for (int i = 0; i < col; ++i) {
50  u(i) = T(0);
51  }
52  T alpha = sign(R(col, col)) * LengthPrecise(u);
53  u(col) -= alpha;
54  T norm = LengthPrecise(u);
55  if (norm == 0) {
56  continue;
57  }
58  u /= norm;
59  v = u;
60  Qi = (T(-2) * v) * Transpose(v);
61  for (int i = 0; i < Q.ColumnCount(); ++i) {
62  Qi(i, i) += T(1);
63  }
64  R = Qi * R;
65  Q = Qi * Q;
66  }
67  Q = Transpose(Q);
68 
69  return DecompositionQR<T, Rows, Columns, Order, Layout, Packed>{ Q, R };
70 }
